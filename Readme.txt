A1 = Sensor ultrasonido-Trigger (Digital)
A2 = Sensor línea izquierda TCRT5000 (Digital)
A3 = Fotorresistencia LDR
A4 = I2C LCD (SCA)
A5 = I2C LCD (SCL)

D0 = (RX) = ** VACIO **
D1 = (TX) = ** VACIO **
D2 = Led blanco derecho
D3-= Led rojo derecho
D4 = Sensor línea derecha TCRT5000 (Digital)
D5-= ENB motor izquierdo
D6-= ENA motor derecho
D7 = IN1 motor derecho
D8 = IN2 motor derecho
D9-= IN3 motor izquierdo
D10-= IN4 motor izquierdo
D11-= Buzzer
D12 = Led rojo izquierdo
D13 = Led blanco izquierdo
